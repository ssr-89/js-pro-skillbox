helpers.js

export function getWindowSize() {
  return {
    height: window.innerHeight,
    width: window.innerWidth
  }
}

size-tracker.js

import { getWindowSize } from "./helpers.js";

console.log('Размер окна', getWindowSize());


export default

syntax.js

function fn() { }
const Constant = '';
let Variable = '';

export { fn as SuperFunction, Constant, Variable }

export default () => {}

import-syntax.js

import { SuperFunction, Constant as C } from "./js/syntax.js";
// переименовать необходимо, чтобы не было конфликтов импортируемых имён

import SomeFunction from './js/syntax.js' // импортирует default из syntax.js

// import SomeFunction, { SuperFunction, Constant as C } from "./js/syntax.js"; - чтобы импортировать всё сразу вначале прописывается дефолтная функция, потом именованные

SuperFunction();
console.log(C);
